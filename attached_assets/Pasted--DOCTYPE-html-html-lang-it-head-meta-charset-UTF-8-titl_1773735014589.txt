<script>
function showPage(id) {
  document.querySelectorAll(".page").forEach(p => p.classList.remove("active"));
  document.getElementById(id).classList.add("active");
}

/* BOARD LOGIC */
const canvas = document.getElementById("canvas");
const ctx = canvas.getContext("2d");

let elements = [];
let tool = "player";
let drawing = false;
let path = [];

function drawField() {
  ctx.fillStyle = "#2e7d32";
  ctx.fillRect(0, 0, canvas.width, canvas.height);

  ctx.strokeStyle = "white";
  ctx.strokeRect(10, 10, canvas.width-20, canvas.height-20);

  ctx.beginPath();
  ctx.moveTo(canvas.width/2, 10);
  ctx.lineTo(canvas.width/2, canvas.height-10);
  ctx.stroke();
}

function drawElements() {
  elements.forEach(el => {
    if (el.type === "player") {
      ctx.fillStyle = "blue";
      ctx.beginPath();
      ctx.arc(el.x, el.y, 10, 0, Math.PI*2);
      ctx.fill();
    }
    if (el.type === "ball") {
      ctx.fillStyle = "white";
      ctx.beginPath();
      ctx.arc(el.x, el.y, 5, 0, Math.PI*2);
      ctx.fill();
    }
    if (el.type === "cone") {
      ctx.fillStyle = "orange";
      ctx.fillRect(el.x-5, el.y-5, 10, 10);
    }
    if (el.type === "path") {
      ctx.strokeStyle = "yellow";
      ctx.beginPath();
      el.points.forEach((p,i)=>{
        if(i===0) ctx.moveTo(p.x,p.y);
        else ctx.lineTo(p.x,p.y);
      });
      ctx.stroke();
    }
  });
}

function render() {
  drawField();
  drawElements();
}

canvas.addEventListener("click", e => {
  const rect = canvas.getBoundingClientRect();
  const x = e.clientX - rect.left;
  const y = e.clientY - rect.top;

  if(tool !== "draw"){
    elements.push({type:tool,x,y});
    render();
  }
});

canvas.addEventListener("mousedown", () => {
  if(tool==="draw"){
    drawing=true;
    path=[];
  }
});

canvas.addEventListener("mousemove", e => {
  if(!drawing) return;
  const rect = canvas.getBoundingClientRect();

  path.push({
    x:e.clientX-rect.left,
    y:e.clientY-rect.top
  });

  render();

  ctx.beginPath();
  path.forEach((p,i)=>{
    if(i===0) ctx.moveTo(p.x,p.y);
    else ctx.lineTo(p.x,p.y);
  });
  ctx.stroke();
});

canvas.addEventListener("mouseup", ()=>{
  if(drawing){
    elements.push({type:"path",points:path});
    drawing=false;
  }
});

function setTool(t){ tool=t }

/* PRESETS */
function preset433(){
  elements = [];
  const positions = [
    [100,250],[250,100],[250,250],[250,400],
    [400,150],[400,350],[550,250],
    [700,100],[700,250],[700,400],[800,250]
  ];
  positions.forEach(p=>elements.push({type:"player",x:p[0],y:p[1]}));
  render();
}

    [700,150],[700,350],[800,250]

/* SAVE MULTI */
function saveTactic(){
  const name = document.getElementById("tacticName").value;
  if(!name) return alert("Inserisci nome");

  let tactics = JSON.parse(localStorage.getItem("tactics") || "[]");
  tactics.push({name, data:elements});
  localStorage.setItem("tactics", JSON.stringify(tactics));

  alert("Salvata!");
}

function loadTactics(){
  const list = document.getElementById("tacticList");
  list.innerHTML="";

  let tactics = JSON.parse(localStorage.getItem("tactics") || "[]");

  tactics.forEach((t,i)=>{
    const btn = document.createElement("button");
    btn.innerText = t.name;
    btn.onclick = ()=>{
      elements = t.data;
      render();
    };
    list.appendChild(btn);
  });
}

render();
